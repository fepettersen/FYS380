\section{Mathematical models}

\begin{frame}
 \frametitle{Basic diffusion}
 The basic diffusion equation reads
 \begin{equation}
  \frac{\d C}{\d t} = D\nabla^2C
 \end{equation}
Einstein relations
\begin{equation}
 D = \frac{k_B T}{6\pi\eta r}
\end{equation}
\begin{equation}
 \langle r^2\rangle = 2dDt
\end{equation}
\end{frame}
